### 3.3 Rust IPC 类型定义

> 所有通过 Tauri IPC 传输的 Rust 结构体必须使用 `#[derive(TS)]` 导出 TypeScript 类型。
> 前端类型由 `ts-rs` 自动生成到 `src-tauri/generated/`，禁止手动编辑。

```rust
use serde::{Deserialize, Serialize};
use ts_rs::TS;
use std::collections::HashMap;

// ─── HTTP 请求类型 ───

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/http.rs.ts")]
pub struct HttpRequestConfig {
    pub method: String,
    pub url: String,
    pub headers: Vec<KvPair>,
    pub params: Vec<KvPair>,
    pub body: Option<RequestBody>,
    pub auth: Option<AuthConfig>,
    pub settings: RequestSettings,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/http.rs.ts")]
pub struct KvPair {
    pub key: String,
    pub value: String,
    pub description: Option<String>,
    pub disabled: bool,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/http.rs.ts")]
#[serde(tag = "mode", content = "data")]
pub enum RequestBody {
    none,
    formdata { data: Vec<FormDataEntry> },
    urlencoded { data: Vec<KvPair> },
    raw { language: String, content: String },
    binary { file_path: String },
    graphql { query: String, variables: String, operation_name: Option<String> },
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/http.rs.ts")]
pub struct FormDataEntry {
    pub key: String,
    pub value: String,
    #[serde(rename = "type")]
    pub entry_type: String, // "text" | "file"
    pub description: Option<String>,
    pub disabled: bool,
    pub content_type: Option<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/http.rs.ts")]
#[serde(tag = "type", content = "config")]
pub enum AuthConfig {
    none { config: EmptyAuth },
    apikey { config: ApiKeyAuth },
    bearer { config: BearerAuth },
    basic { config: BasicAuth },
    jwt { config: JwtAuth },
    oauth1 { config: OAuth1Auth },
    oauth2 { config: OAuth2Auth },
    awsv4 { config: AwsV4Auth },
    digest { config: EmptyAuth },
    ntlm { config: EmptyAuth },
    hawk { config: EmptyAuth },
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct EmptyAuth {}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct ApiKeyAuth {
    pub key: String,
    pub value: String,
    pub add_to: String, // "header" | "query"
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct BearerAuth {
    pub token: String,
    pub prefix: Option<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct BasicAuth {
    pub username: String,
    pub password: String,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct JwtAuth {
    pub payload: String,
    pub secret: String,
    pub algorithm: String,
    pub add_to: String,
    pub header_prefix: String,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct OAuth1Auth {
    pub consumer_key: String,
    pub consumer_secret: String,
    pub token: String,
    pub token_secret: String,
    pub signature_method: String,
    pub version: String,
    pub realm: Option<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct OAuth2Auth {
    pub grant_type: String,
    pub auth_url: Option<String>,
    pub access_token_url: Option<String>,
    pub client_id: Option<String>,
    pub client_secret: Option<String>,
    pub scope: Option<String>,
    pub state: Option<String>,
    pub redirect_uri: Option<String>,
    pub pkce: Option<bool>,
    pub access_token: Option<String>,
    pub refresh_token: Option<String>,
    pub expires_in: Option<u64>,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct AwsV4Auth {
    pub access_key: String,
    pub secret_key: String,
    pub session_token: Option<String>,
    pub service: String,
    pub region: String,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct RequestSettings {
    pub timeout_ms: u64,
    pub max_redirects: u32,
    pub follow_redirects: bool,
    pub skip_ssl_verification: bool,
    pub proxy: Option<ProxyConfig>,
    pub keep_alive: bool,
    pub send_cookies: bool,
    pub store_cookies: bool,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct ProxyConfig {
    pub proxy_type: String, // "http" | "socks5" | "system"
    pub url: String,
    pub auth: Option<ProxyAuth>,
    pub bypass: Vec<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct ProxyAuth {
    pub username: String,
    pub password: String,
}

// ─── HTTP 响应类型 ───

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/http.rs.ts")]
pub struct HttpResponse {
    pub request_id: String,
    pub status: u16,
    pub status_text: String,
    pub headers: Vec<KvPair>,
    pub body: String,             // Base64 编码（二进制响应）或 UTF-8 文本
    pub body_is_binary: bool,     // true 时 body 为 Base64 编码
    pub body_size: u64,
    pub content_type: String,
    pub encoding: String,
    pub time_ms: u64,
    pub timeline: Option<ResponseTimeline>,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct ResponseTimeline {
    pub dns_ms: u64,
    pub tcp_ms: u64,
    pub tls_ms: u64,
    pub first_byte_ms: u64,
    pub content_download_ms: u64,
    pub total_ms: u64,
}

// ─── gRPC 请求类型 ───

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
#[ts(export_to = "generated/grpc.rs.ts")]
pub struct GrpcRequestConfig {
    pub url: String,
    pub service: String,
    pub method: String,
    pub message: String,
    pub metadata: HashMap<String, String>,
    pub proto_file_path: Option<String>,
    pub deadline_ms: Option<u64>,
    pub use_tls: bool,
}

#[derive(Debug, Clone, Serialize, Deserialize, TS)]
pub struct GrpcResponse {
    pub status: String,
    pub status_code: i32,
    pub headers: HashMap<String, String>,
    pub message: String,
    pub time_ms: u64,
```

> 1. **`HttpResponse.body` 为 String**：二进制响应使用 Base64 编码，通过 `body_is_binary` 标记区分。避免 IPC 传输 `Vec<u8>` 的序列化问题。

### 3.5 SSE 模块设计

SSE (Server-Sent Events) 连接通过 Rust 后端处理，使用 reqwest 的流式响应：

```rust
#[tauri::command]
pub async fn sse_connect(
    state: tauri::State<'_, AppState>,
    url: String,
    headers: HashMap<String, String>,
    request_id: String,
) -> Result<String, String> {
    let client = &state.http_client.client;
    let mut request = client.get(&url);
    for (k, v) in headers {
        request = request.header(&k, &v);
    }

    let response = request.send().await.map_err(|e| e.to_string())?;

    if !response.status().is_success() {
        return Err(format!("SSE connection failed: {}", response.status()));
    }

    let app = state.app_handle.clone();
    let stream = response.bytes_stream();

    tokio::spawn(async move {
        let mut stream = Box::pin(stream);
        use futures_util::StreamExt;

        while let Some(chunk) = stream.next().await {
            match chunk {
                Ok(bytes) => {
                    let data = String::from_utf8_lossy(&bytes).to_string();
                    let _ = app.emit("sse-event", &SsePayload {
                        request_id: request_id.clone(),
                        data,
                        event_type: "message".to_string(),
                    });
                }
                Err(e) => {
                    let _ = app.emit("sse-error", &SsePayload {
                        request_id: request_id.clone(),
                        data: e.to_string(),
                        event_type: "error".to_string(),
                    });
                    break;
                }
            }
        }

        let _ = app.emit("sse-close", &SsePayload {
            request_id: request_id.clone(),
            data: String::new(),
            event_type: "close".to_string(),
        });
    });

    Ok(request_id)
}

#[derive(Clone, Serialize)]
struct SsePayload {
    request_id: String,
    data: String,
    event_type: String,
}

#[tauri::command]
pub async fn sse_disconnect(
    state: tauri::State<'_, AppState>,
    request_id: String,
) -> Result<(), String> {
    // 通过 cancel_tokens 取消 SSE 流
    let tokens = state.cancel_tokens.read().await;
    if let Some(token) = tokens.get(&request_id) {
        token.cancel();
    }
    Ok(())
}
```

### 4.5 Tab 与 Request 关系模型

Tab 是前端 UI 概念，Request 是业务数据概念。两者的映射关系决定了编辑和持久化行为。

```typescript
interface TabState {
  id: string;                        // Tab 唯一 ID（前端生成）
  requestId: string | null;          // 关联的集合请求 ID（null = 临时请求）
  collectionId: string | null;       // 所属集合 ID
  name: string;                      // Tab 显示名称
  method: string;                    // HTTP 方法
  url: string;                       // 请求 URL

  // 编辑状态
  isDirty: boolean;                  // 编辑后未保存
  dirtyFields: Set<string>;          // 具体修改的字段
  lastSavedHash: string;             // 上次保存时的内容哈希

  // 编辑副本：每个 Tab 持有独立的请求配置副本
  requestCopy: HttpRequestConfig;    // Tab 级请求配置快照
}
```

**关键设计决策**：

| 决策 | 选择 | 原因 |
|------|------|------|
| 同一 Request 能否在多个 Tab 打开？ | **否** | 避免多 Tab 编辑同一请求的冲突；打开时检查是否已有同 requestId 的 Tab |
| 临时请求如何处理？ | `requestId: null` | 不保存到集合；关闭 Tab 时数据丢失（dirty 时弹确认） |
| Tab 切换时是否保存？ | 自动保存当前 Tab | 切换前自动保存到文件系统，避免数据丢失 |
| 响应数据是否持久化？ | 否 | 响应仅存在于 `RequestStore.responses[tabId]`，重启后需重新发送 |

**Tab 打开流程**：

```
用户点击集合请求 / 历史记录
  │
  ├─ 1. 检查 TabStore 中是否已有同 requestId 的 Tab
  │     ├── 已存在 → setActiveTab(tabId)
  │     └── 不存在 → 继续
  │
  ├─ 2. invoke('get_request', { requestId }) 加载完整请求数据
  │
  ├─ 3. 创建 TabState { requestId, requestCopy: deepClone(requestData) }
  │
  ├─ 4. TabStore.openTab(tabState)
  │
  └─ 5. RequestStore.setActiveTab(tabId)
```

**Tab 关闭流程**：

```
用户点击 Tab 关闭按钮
  │
  ├─ Tab dirty？
  │     ├── 是 → 弹出 Save / Don't Save / Cancel
  │     │         ├── Save → invoke('write_file', ...) → 继续关闭
  │     │         ├── Don't Save → 继续关闭（丢失编辑）
  │     │         └── Cancel → 取消关闭
  │     └── 否 → 继续关闭
  │
  ├─ RequestStore.clearResponse(tabId)
  │
  └─ TabStore.closeTab(tabId)
```
