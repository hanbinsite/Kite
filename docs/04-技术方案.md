### 7.2 OAuth 流程

```typescript
export class OAuth2Service {
  async startAuthCodeFlow(
    config: OAuth2Auth,
  ): Promise<OAuth2TokenResponse> {
    const codeVerifier = generateRandomString(128);
    const codeChallenge = await sha256Base64(codeVerifier);
    const state = generateRandomString(32);

    const authUrl = new URL(config.authUrl!);
    authUrl.searchParams.set('response_type', 'code');
    authUrl.searchParams.set('client_id', config.clientId!);
    authUrl.searchParams.set(
      'redirect_uri',
      'apiclient://oauth/callback',
    );
    authUrl.searchParams.set('scope', config.scope ?? '');
    authUrl.searchParams.set('state', state);
    if (config.pkce) {
      authUrl.searchParams.set('code_challenge', codeChallenge);
      authUrl.searchParams.set(
        'code_challenge_method',
        'S256',
      );
    }

    await openExternal(authUrl.toString());

    return new Promise((resolve, reject) => {
      const timeout = setTimeout(
        () => reject(new Error('OAuth timeout')),
        300000,
      );

      const unlisten = listen(
        'oauth-callback',
        (event) => {
          clearTimeout(timeout);
          unlisten.then((fn) => fn());

          const params = new URLSearchParams(
            event.payload as string,
          );
          if (params.get('state') !== state) {
            reject(new Error('State mismatch'));
            return;
          }

          const code = params.get('code');
          if (!code) {
            reject(new Error('No code in callback'));
            return;
          }

          this.exchangeCodeForToken(code, config, codeVerifier)
            .then(resolve)
            .catch(reject);
        },
      );
    });
  }

  private async exchangeCodeForToken(
    code: string,
    config: OAuth2Auth,
    codeVerifier: string,
  ): Promise<OAuth2TokenResponse> {
    return invoke('exchange_oauth_token', {
      code,
      tokenUrl: config.accessTokenUrl,
      clientId: config.clientId,
      clientSecret: config.clientSecret,
      redirectUri: 'apiclient://oauth/callback',
      codeVerifier,
    });
  }
}
```

## 8. 性能优化

### 8.1 性能目标

| 指标 | 目标 | 测量方法 |
|------|------|----------|
| 冷启动 | < 2秒 | 从点击图标到界面可用 |
| 热启动 | < 500ms | 应用内操作响应 |
| 内存(空闲) | < 150MB | 打开应用不操作 |
| 内存(工作) | < 400MB | 正常使用 |
| 请求发送延迟 | < 50ms | 本地请求 |
| 响应渲染 | < 100ms | 1MB JSON |

### 8.2 优化策略

| 策略 | 实现 |
|------|------|
| 代码分割 | React.lazy + Suspense |
| Monaco 延迟加载 | 首次打开编辑器时加载 |
| Monaco 单实例 | 切换 Tab 时替换 Model，不创建新实例 |
| 轻量编辑器 | URL/Headers 用 CodeMirror 6，JSON/JS 用 Monaco |
| 虚拟列表 | @tanstack/react-virtual |
| 防抖/节流 | ts-debounce / throttleit |
| Web Worker | 大 JSON 解析/格式化放入 Worker |
| 流式响应 | Rust 侧流式返回大响应体 |
| SQLite 索引 | 历史记录、Cookie 等结构化查询建索引 |
| 骨架屏 | Loading skeletons |
| 字体预加载 | 本地打包 Geist, JetBrains Mono |
| Rust 预初始化 | 应用启动时即初始化 HTTP 客户端池 |

### 3.3 脚本引擎实现

```rust
use rquickjs::{Context, Function, Runtime, Undefined};
use std::time::Duration;

pub struct ScriptEngine {
    runtime: Runtime,
}

impl ScriptEngine {
    pub fn new() -> Result<Self, String> {
        let runtime = Runtime::new().map_err(|e| e.to_string())?;
        runtime.set_memory_limit(128 * 1024 * 1024); // 128MB
        runtime.set_max_stack_size(1024 * 1024);     // 1MB
        Ok(Self { runtime })
    }

    pub fn execute(
        &self,
        code: &str,
        context: ScriptContext,
        timeout_ms: u64,
    ) -> Result<ScriptResult, String> {
        let ctx = Context::full(&self.runtime).map_err(|e| e.to_string())?;

        let result = ctx.with(|ctx| {
            let pm = create_pm_object(&ctx, &context)?;
            ctx.globals().set("pm", pm)?;

            let console = create_console_object(&ctx)?;
            ctx.globals().set("console", console)?;

            let result: Result<Undefined, rquickjs::Error> = ctx.eval(code);

            match result {
                Ok(_) => Ok(ScriptResult {
                    success: true,
                    logs: get_captured_logs(),
                    variables: get_modified_variables(),
                    error: None,
                }),
                Err(e) => Ok(ScriptResult {
                    success: false,
                    logs: get_captured_logs(),
                    variables: get_modified_variables(),
                    error: Some(format!("{:?}", e)),
                }),
            }
        });

        result
    }
}

fn create_pm_object<'js>(
    ctx: &rquickjs::Ctx<'js>,
    context: &ScriptContext,
) -> Result<rquickjs::Object<'js>, String> {
    let pm = rquickjs::Object::new(ctx.clone()).map_err(|e| e.to_string())?;

    let request = rquickjs::Object::new(ctx.clone()).map_err(|e| e.to_string())?;
    request.set("method", &context.request.method).map_err(|e| e.to_string())?;
    request.set("url", &context.request.url).map_err(|e| e.to_string())?;
    pm.set("request", request).map_err(|e| e.to_string())?;

    if let Some(ref response) = context.response {
        let resp = rquickjs::Object::new(ctx.clone()).map_err(|e| e.to_string())?;
        resp.set("status", response.status).map_err(|e| e.to_string())?;
        resp.set("body", &response.body).map_err(|e| e.to_string())?;
        pm.set("response", resp).map_err(|e| e.to_string())?;
    }

    let variables = rquickjs::Object::new(ctx.clone()).map_err(|e| e.to_string())?;
    let vars_data = context.variables.clone();
    variables
        .set(
            "get",
            Function::new(ctx.clone(), move |name: String| -> Option<String> {
                vars_data.get(&name).cloned()
            }),
        )
        .map_err(|e| e.to_string())?;
    pm.set("variables", variables).map_err(|e| e.to_string())?;

    let test_fn = Function::new(ctx.clone(), |name: String, _fn_val: rquickjs::Function| {
        // 测试执行逻辑
    })
    .map_err(|e| e.to_string())?;
    pm.set("test", test_fn).map_err(|e| e.to_string())?;

    Ok(pm)
}
```

### 3.4 Vault 加密实现

```rust
use keyring::Entry;
use sodiumoxide::crypto::secretbox;
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize)]
struct StoredSecret {
    name: String,
    nonce: String,
    ciphertext: String,
    updated_at: String,
}

pub struct VaultService {
    key: Option<secretbox::Key>,
    base_path: std::path::PathBuf,
}

impl VaultService {
    pub fn new(base_path: std::path::PathBuf) -> Self {
        Self {
            key: None,
            base_path,
        }
    }

    pub fn unlock(&mut self, master_password: &str) -> Result<(), String> {
        let derived_key = self.derive_key(master_password)?;
        self.key = Some(derived_key);

        let entry =
            Entry::new("api-client", "vault-key").map_err(|e| e.to_string())?;
        entry
            .set_password(&hex::encode(self.key.as_ref().unwrap().as_ref()))
            .map_err(|e| e.to_string())?;

        Ok(())
    }

    pub fn lock(&mut self) -> Result<(), String> {
        self.key = None;
        let entry =
            Entry::new("api-client", "vault-key").map_err(|e| e.to_string())?;
        let _ = entry.delete_credential();
        Ok(())
    }

    pub fn is_unlocked(&self) -> bool {
        self.key.is_some()
    }

    pub fn encrypt_secret(&self, name: &str, plaintext: &str) -> Result<(), String> {
        let key = self.key.as_ref().ok_or("Vault is locked")?;
        let nonce = secretbox::gen_nonce();
        let ciphertext = secretbox::seal(plaintext.as_bytes(), nonce, key);

        let stored = StoredSecret {
            name: name.to_string(),
            nonce: hex::encode(nonce.as_ref()),
            ciphertext: hex::encode(&ciphertext),
            updated_at: chrono::Utc::now().to_rfc3339(),
        };

        let vault_path = self.get_secret_path(name);
        let json =
            serde_json::to_string_pretty(&stored).map_err(|e| e.to_string())?;
        std::fs::write(&vault_path, json).map_err(|e| e.to_string())?;

        Ok(())
    }

    pub fn decrypt_secret(&self, name: &str) -> Result<String, String> {
        let key = self.key.as_ref().ok_or("Vault is locked")?;

        let vault_path = self.get_secret_path(name);
        let json = std::fs::read_to_string(&vault_path).map_err(|e| e.to_string())?;
        let stored: StoredSecret =
            serde_json::from_str(&json).map_err(|e| e.to_string())?;

        let nonce_bytes = hex::decode(&stored.nonce).map_err(|e| e.to_string())?;
        let nonce = secretbox::Nonce::from_slice(&nonce_bytes).ok_or("Invalid nonce")?;

        let ciphertext_bytes =
            hex::decode(&stored.ciphertext).map_err(|e| e.to_string())?;

        let plaintext = secretbox::open(&ciphertext_bytes, &nonce, key)
            .map_err(|_| "Decryption failed".to_string())?;

        String::from_utf8(plaintext).map_err(|e| e.to_string())
    }

    fn derive_key(&self, password: &str) -> Result<secretbox::Key, String> {
        use sodiumoxide::crypto::pwhash;

        let salt = pwhash::gen_salt();
        let mut key = [0u8; secretbox::KEYBYTES];

        pwhash::derive_key(
            &mut key,
            password.as_bytes(),
            &salt,
            pwhash::OPSLIMIT_INTERACTIVE,
            pwhash::MEMLIMIT_INTERACTIVE,
        )
        .map_err(|e| e.to_string())?;

        Ok(secretbox::Key::from_slice(&key).unwrap())
    }

    fn get_secret_path(&self, name: &str) -> std::path::PathBuf {
        self.base_path.join("vault").join(format!("{}.enc.json", name))
    }
}
```

### 4.3 错误边界设计

```typescript
import { Component, ReactNode, ErrorInfo } from 'react';

interface ErrorBoundaryProps {
  children: ReactNode;
  fallback?: ReactNode;
}

interface ErrorBoundaryState {
  hasError: boolean;
  error: Error | null;
}

export class ErrorBoundary extends Component<
  ErrorBoundaryProps,
  ErrorBoundaryState
> {
  constructor(props: ErrorBoundaryProps) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('ErrorBoundary caught:', error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback ?? (
        <div className="p-4 text-center">
          <p className="text-red-500">Something went wrong</p>
          <button
            onClick={() =>
              this.setState({ hasError: false, error: null })
            }
            className="mt-2 px-4 py-2 bg-accent text-white rounded"
          >
            Retry
          </button>
        </div>
      );
    }
    return this.props.children;
  }
}
```

```typescript
  createModel(tabId: string, content: string, language: string) {
    const uri = monaco.Uri.parse(`inmemory://${tabId}`);
    let model = monaco.editor.getModel(uri);
    if (!model) {
      model = monaco.editor.createModel(content, language, uri);
    }
    this.models.set(tabId, model);
    return model;
  }

  switchToTab(tabId: string) {
    const model = this.models.get(tabId);
    if (model && this.editor) {
      this.editor.setModel(model);
    }
  }

  disposeModel(tabId: string) {
    const model = this.models.get(tabId);
    if (model) {
      model.dispose();
      this.models.delete(tabId);
    }
  }
}
```

```typescript
export class HttpService {
  async send(request: HttpRequest): Promise<HttpResponse> {
    const resolvedRequest = await this.resolveVariables(request);

    if (resolvedRequest.preRequestScript) {
      const scriptResult = await executeScript(
        resolvedRequest.preRequestScript,
        {
          request: resolvedRequest,
          variables: this.getVariables(),
        },
        5000,
      );
      this.applyScriptResult(scriptResult);
    }

    const response = await sendHttpRequest(resolvedRequest);

    if (resolvedRequest.postResponseScript) {
      const scriptResult = await executeScript(
        resolvedRequest.postResponseScript,
        {
          request: resolvedRequest,
          response,
          variables: this.getVariables(),
        },
        5000,
      );
      this.applyScriptResult(scriptResult);
    }

    return response;
  }

  async cancel(requestId: string): Promise<void> {
    await cancelHttpRequest(requestId);
  }

  private async resolveVariables(
    request: HttpRequest,
  ): Promise<HttpRequest> {
    // 变量替换逻辑
  }

  private applyScriptResult(result: ScriptResult): void {
    if (result.variables) {
      for (const [key, value] of Object.entries(result.variables)) {
        useEnvironmentStore.getState().setVariable(key, value);
      }
    }
  }

  private getVariables(): Record<string, string> {
    const env = useEnvironmentStore.getState();
    const vars: Record<string, string> = {};
    for (const v of env.globals) {
      if (!v.disabled) vars[v.key] = v.value;
    }
    const activeEnv = env.environments.find(
      (e) => e.id === env.activeEnvironmentId,
    );
    if (activeEnv) {
      for (const v of activeEnv.values) {
        if (v.enabled) vars[v.key] = v.value;
      }
    }
    return vars;
  }
}
```
